# Firebase Storage Security Rules
# Copy these rules to your Firebase Console > Storage > Rules

# DEVELOPMENT RULES (More permissive - use these for now)
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /{allPaths=**} {
      allow read, write: if true;
    }
  }
}
